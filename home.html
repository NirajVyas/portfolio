class Niraj < WebDeveloper::Base
  acts_as_developer
  attr_accessor :Ruby, :Javascript, :HTML5, :CSS3, :Git

  has_many :skills
  belongs_to :WebDeveloper

  def role?(role)
    self.junior.web.developer = role.new
  end

  def previous?(role)
    self.graphic.designer = role
  end

  def teamPlayer?
    attr_accessor :respectful, :communicator, :friendly
  end

  module SkillLevel
    Expert = 4
    Advanced = 3
    Intermediate = 2
    Beginner = 1
    Undefined = nil
  end

  def programmer 
    @programmer.each do |skill|
      skill.ruby.level == 2
      skill.javascript.level == 2
      skill.HTML5.level == 3
      skill.CSS3.level == 3
  end

  def software
    @software.each do |skill|
      skill.github.level == 2
      skill.terminal.level == 2
      skill.photoshop.level == 3
      skill.3DsMax.level == 3
      skill.sublime.level == 3
      skill.heroku.level == 3
      skill.amazonS3.level == 3
  end

  def API
    @API.each do |API|
      API.Google.Maps.level == 3
      API.YouTube.level == 3
      API.Facebook.level == 2
      API.Twitter.level == 2
  end

  def education
    @generalAssembly = true {12 weeks immersive}
    @university = true {1st class honours degree}
    @collge = true {Distinction, Distinction, Merit (Multimedia)}
    @secondaryScho{ 10+ GCSE A* - C }
  end
end
